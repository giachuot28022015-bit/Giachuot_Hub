-- TRÌNH PHÁT NHẠC (ĐÃ TỐI ƯU LỆNH)
MusicTab:CreateSection("Trình phát nhạc")
local MusicId = ""
MusicTab:CreateInput({Name = "ID Nhạc", PlaceholderText = "Nhập ID...", Callback = function(Value) MusicId = Value end})
MusicTab:CreateButton({Name = "▶️ Phát Nhạc", Callback = function()
    local srv = game:GetService("SoundService")
    for _, s in pairs(srv:GetChildren()) do if s.Name == "GiaChuotMusic" then s:Stop() s:Destroy() end end
    local sound = Instance.new("Sound") sound.Name = "GiaChuotMusic" sound.SoundId = "rbxassetid://" .. MusicId sound.Parent = srv sound:Play()
end})
MusicTab:CreateButton({Name = "⏹️ Dừng Nhạc", Callback = function()
    for _, s in pairs(game:GetService("SoundService"):GetChildren()) do if s.Name == "GiaChuotMusic" then s:Stop() s:Destroy() end end
end})
